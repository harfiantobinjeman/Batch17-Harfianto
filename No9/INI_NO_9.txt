	A.SELECT * FROM T_POLICY P
		JOIN T_CLIENT C ON C.CLIENT_NUMBER = P.CLIENT_NUMBER
		WHERE P.POLICE_SUBMIT_DATE BETWEEN '15.01.2018' AND '01.02.2018'
		
	B.SELECT P.POLICE_NUMBER, C.AGENT_NAME,P.POLICE_STATUS,
		C.AGENT_OFFICE FROM T_POLICE P
		JOIN T_AGENT A ON A.AGENT_CODE = P.AGENT_CODE
		GROUP BY P.POLICE_NUMBER, C.AGENT_NAME,P.POLICE_STATUS,
		C.AGENT_OFFICE
	C.SELECT A.AGENT.CODE, AGENT_NAME,A.BASIC_COMMISSION(SELECT
		(P.COMMISSION / P.PREMIUM) * 100 FROM T_AGENT A
		JOIN T_POLICY P ON P.AGENT.CODE = A.AGENT.CODE
		GROUP BY A.AGENT.CODE, AGENT_NAME)
		FROM T_AGENT A
		JOIN T_POLICY P ON P.AGENT.CODE = A.AGENT.CODE
		GROUP BY A.AGENT.CODE, AGENT_NAME,A.BASIC_COMMISSION
	D.INSERT INTO T_POLICY (POLICY_DUE_DATE)
		VALUE (ADDDATE(SELECT POLICY_SUBMIT_DATE FROM T_POLICY)
			,INTERVAL '1' MOUNTH)
	E.SELECT * FROM T_POLICY
		WHERE PREMIUM = (SELECT ((PREMIUM * DISCOUNT)/100) 
		AS DISCOUNT FROM T_POLICY)
		ORDER BY PREMIUM ASC
